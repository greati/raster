45  std::vector<EdgeEntry> general_edges[_canvas.height()];

47  // create general table with all vertices (ET), ordered ascending col_intercept
48  for (auto & [obj_name, polygon] : polygons) {
49 
50  if (polygon.fill() == std::nullopt) continue;
51 
52  // collect edges from polygons and annotate important info
53  auto vertices = polygon.vertices();
54  for (auto i = 0; i < vertices.size() - 1; ++i) {
55  auto [p1x, p1y] = vertices[i];
56  auto [p2x, p2y] = vertices[i + 1];
57  if (p1x > p2x) {
58  std::swap(p1x, p2x);
59  std::swap(p1y, p2y);
60  }
61  auto min_x = std::lround(std::min(p1x, p2x)); // when it has appeared
62  auto max_x = std::lround(std::max(p1x, p2x)); // when will it disappear
63  // ignore horizontal
64  if (std::lround(p2x-p1x) == 0) continue;
65  EdgeEntry edge {
66  obj_name,
67  max_x,
68  std::lround(p1y),
69  std::lround(p2x-p1x),
70  std::lround(p2y-p1y)
71  };
72  general_edges[min_x].push_back(edge);
73  }
74  }

77  return a._row_max < b._row_max;
78  };

81  return a._col_intercept < b._col_intercept;
82  };

85  for (int i = 0; i < _canvas.height(); ++i) {
86  std::sort(general_edges[i].begin(), general_edges[i].end(), sort_inc_rowmax);
87  }

92  while (i < _canvas.height()) {

97  active_edges.sort(sort_col_interc);
98  for (auto it = active_edges.begin(); it != active_edges.end();) {
99  auto e1 = *it;
100  if (++it != active_edges.end()) {
101  auto e2 = *it;
102  if (e1._col_intercept == e2._col_intercept) {
103  if (i == e1._row_max) continue;
104  else if(i == e2._row_max) {
105  if (++it != active_edges.end())
106  e2 = *it;
107  else break;
108  }
109  }
110  auto [r, g, b] = polygons.find(e1._obj_name)->second.fill().value().color;
111  for (int x = e1._col_intercept; x < e2._col_intercept; ++x) {
112  this->_canvas.set({i, x}, {r, g, b});
113  }
114  if (++it != active_edges.end()) ;
115  else break;
116  } else break;
117  }
118  // erase row_max edges

120  if (it->_row_max == i) {
121  it = active_edges.erase(it);
122  } else it++;
123  }
124  // next line
125  i++;
126  // update cols
127  for (auto edge = active_edges.begin(); edge != active_edges.end(); ++edge) {
128  if (edge->_delta_row != 0) { // non-horizontal
129  if (edge->_delta_row > 0) { // down
130  long add = edge->_slope_int +
131  ((edge->_remainder_col + edge->_slope_frac) / edge->_delta_row);
132  edge->_remainder_col = (edge->_remainder_col + edge->_slope_frac) % edge->_delta_row;
133  edge->_col_intercept += add;//;
134  } else { // up

137  edge->_remainder_col = (edge->_remainder_col + edge->_slope_frac) % std::abs(edge->_delta_row);
138  edge->_col_intercept -= add;//;
139  }
